URL: /login

    "TeamID" : int,
    "TeamName" : string,
    "RoleName" : string
}

URL: /file/$fileID
method: GET
param: 路由中就有的$fileID以及用cookie传递的UserID、UserPassword
return 一个文件，文件类型可以取：图片、Office全家桶的各种格式的文件、压缩包

URL: /file

            "fileType" : string,

}
